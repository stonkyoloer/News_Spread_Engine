import pandas as pd
import yfinance as yf

# Define the selection criteria
liquidity_threshold = 1000000
price_range = (50, 500)
sector_limit = 3
volatility_profile = ['low', 'medium', 'high']
options_availability = True
market_cap_threshold = 50000000000

# Get the list of S&P 500 stocks
sp500 = pd.read_html('https://en.wikipedia.org/wiki/List_of_S%26P_500_companies')[0]
sp500['Ticker symbol'] = sp500['Ticker symbol'].apply(lambda x: x.replace('.', '-'))

# Initialize the output variables
stocks = []
edge_reason = {}

# Loop through the stocks and apply the selection criteria
for ticker in sp500['Ticker symbol']:
    stock = yf.Ticker(ticker)
    if (stock.info['averageDailyVolume10Day'] > liquidity_threshold and
        price_range[0] <= stock.info['regularMarketPrice'] <= price_range[1] and
        stocks.count(stock.info['sector']) < sector_limit and
        stock.info['impliedSharesOutstanding'] in volatility_profile and
        options_availability and
        stock.info['marketCap'] > market_cap_threshold):
        stocks.append(ticker)
        edge_reason[ticker] = f"{stock.info['sector']} - {stock.info['longBusinessSummary']}"

    # Stop when we have exactly 22 stocks
    if len(stocks) == 22:
        break

# Output the result
print('STOCKS =', stocks)
print('EDGE_REASON =', edge_reason)
